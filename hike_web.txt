  <script>
    let trails = [];
    let currentFilter = "All";
    let currentSearch = "";

    function renderTable() {
      const tableBody = document.getElementById("trailTable");
      tableBody.innerHTML = "";

      let filtered = currentFilter === "All"
        ? trails
        : trails.filter(t => t.Area === currentFilter);

      if (currentSearch.trim() !== "") {
        filtered = filtered.filter(t =>
          t.Name.toLowerCase().includes(currentSearch.toLowerCase())
        );
      }

      if (filtered.length === 0) {
        tableBody.innerHTML = `<tr><td colspan="3">No trails found.</td></tr>`;
        return;
      }

      filtered.forEach(t => {
        const row = document.createElement("tr");
        row.innerHTML = `
          <td>${t.Name}</td>
          <td><a href="${t.Website}" target="_blank">Visit</a></td>
          <td>${t.Area}</td>
        `;
        tableBody.appendChild(row);
      });
    }

    function updateFilterOptions() {
      const filterSelect = document.getElementById("areaFilter");
      const areas = ["All", ...new Set(trails.map(t => t.Area))];
      filterSelect.innerHTML = "";
      areas.forEach(area => {
        const opt = document.createElement("option");
        opt.value = area;
        opt.textContent = area;
        filterSelect.appendChild(opt);
      });
    }

    function loadExcelFile(file, isLocal = false) {
      if (isLocal) {
        const reader = new FileReader();
        reader.onload = evt => processExcel(evt.target.result);
        reader.readAsArrayBuffer(file);
      } else {
        fetch(file)
          .then(res => res.arrayBuffer())
          .then(buffer => processExcel(buffer))
          .catch(err => console.error("Error loading Excel:", err));
      }
    }

    function processExcel(buffer) {
      const workbook = XLSX.read(buffer, { type: 'array' });
      const sheet = workbook.Sheets[workbook.SheetNames[0]];
      trails = XLSX.utils.sheet_to_json(sheet);
      updateFilterOptions();
      renderTable();
    }

    document.getElementById("areaFilter").addEventListener("change", e => {
      currentFilter = e.target.value;
      renderTable();
    });

    document.getElementById("searchBox").addEventListener("input", e => {
      currentSearch = e.target.value;
      renderTable();
    });

    document.getElementById("upload").addEventListener("change", e => {
      const file = e.target.files[0];
      if (file) loadExcelFile(file, true);
    });

    loadExcelFile("trails.xlsx");
  </script>
